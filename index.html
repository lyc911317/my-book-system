自動寫入 Google 試算表 (Google Sheets) 設定

本系統支援遠端同步。您可以透過 Google Apps Script 佈署一個專屬的 Webhook API 網址，當您在手機網頁上完成掃描或新增時，資料就會同時「隔空」寫入您雲端試算表中！

Google 試算表後端設定步驟教學

  1. 建立一個全新的 Google 試算表。
  2. 將第一列設定為標題，請務必在 A1 ~ G1 的格子依序填入以下欄位名：
    ID
    時間
    ISBN
    書名
    作者
    出版日期
    分類
  3. 在試算表頂部選單選擇 「擴充功能」 ➔ 「Apps Script」。
  4. 將後台原本的預設程式碼清空，並完整複製貼上方的 ** Apps Script 程式碼**。
  5. 點選右上角 「部署」 ➔ 「新增部署作業」。
  6. 點擊左側小齒輪，將類型選擇為 「網頁應用程式 (Web App)」。
  7. 將「誰能存取」改為 「任何人 (Anyone)」（此項非常關鍵，否則 iPhone 將因安全限制拒絕存取）。點擊部署。
  8. 將部署成功後取得的 「網頁應用程式網址」 貼至本頁上方的欄位，即完成智慧連線！

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  
  try {
    // 解析前端傳來的書籍資料
    var data = JSON.parse(e.postData.contents);
    
    // 檢查是否有傳入 ISBN
    if (!data.isbn) {
      return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": "缺少 ISBN 欄位。" }))
                           .setMimeType(ContentService.MimeType.JSON);
    }
    
    var lastRow = sheet.getLastRow();
    var targetRow = -1;
    
    // 1. 條件搜尋：在第 3 欄 (C欄) 搜尋 ISBN 是否存在且第 4 欄 (D欄) 書名為空
    for (var i = 2; i <= lastRow; i++) {
      var colCValue = sheet.getRange(i, 3).getValue().toString().trim(); // C欄: ISBN
      var colDValue = sheet.getRange(i, 4).getValue().toString().trim(); // D欄: 書名
      
      if (colCValue === data.isbn && colDValue === "") {
        targetRow = i; // 找到此列
        break;
      }
    }
    
    // 2. 如果沒找到，且最後一列沒有資料，就攔截提示；若有資料則自動建立新一列
    if (targetRow === -1) {
      var checkLastRowIsbn = lastRow > 1 ? sheet.getRange(lastRow, 3).getValue().toString().trim() : "";
      if (checkLastRowIsbn === "" && lastRow > 1) {
        // 最後一列沒有任何 ISBN 碼，代表是空列，直接使用該列
        targetRow = lastRow;
      } else {
        targetRow = lastRow + 1; // 新增一列
      }
    }
    
    // 3. 判斷是否需要生成 ID：新一列才填入隨機不重複 ID (滿足 AppSheet 的 Key 要求)
    if (targetRow > lastRow) {
      var uniqueId = "BK-" + Utilities.getUuid().substring(0, 8).toUpperCase();
      sheet.getRange(targetRow, 1).setValue(uniqueId); // A欄：ID
    }
    
    // 4. 寫入書籍詳細資訊（欄位向右移一欄，完美滿足您的試算表架構）
    sheet.getRange(targetRow, 2).setValue(new Date());          // B欄：寫入時間
    sheet.getRange(targetRow, 3).setValue(data.isbn);           // C欄：ISBN
    sheet.getRange(targetRow, 4).setValue(data.title);          // D欄：書名
    sheet.getRange(targetRow, 5).setValue(data.author);         // E欄：作者
    sheet.getRange(targetRow, 6).setValue(data.publication);    // F欄：出版日期
    sheet.getRange(targetRow, 7).setValue(data.category);       // G欄：主題分類
    
    // 回傳成功狀態給網頁前端
    return ContentService.createTextOutput(JSON.stringify({ "status": "success", "row": targetRow }))
                         .setMimeType(ContentService.MimeType.JSON);
                         
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": err.message }))
                         .setMimeType(ContentService.MimeType.JSON);
  }
}
